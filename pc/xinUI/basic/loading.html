# Loading 加载条

# 引用

import { Loading } from 'xin-ui'

# 示例

# 参数

# status

  • 类型：String
  • 默认值：""
  • 描述：加载条状态。

# 事件

# start

  • 描述：开始加载。
  • 回调参数: 无。

# success

  • 描述：加载成功。
  • 回调参数: 无。

# error

  • 描述：加载失败。
  • 回调参数: 无。

# 依赖情况

无依赖。

# 代码示例

<template>
  <div class="example-page">
    <div class="line">
      <xin-label width="60">Bar 展示</xin-label>
      <xin-button @click="start()">开始加载</xin-button>
      <xin-button color="success" @click="success()">加载成功</xin-button>
      <xin-button color="error" @click="error()">加载失败</xin-button>
    </div>
    <div class="line">
      <!-- <xin-loading
        mode="bar"
        :status="status"
      >
      </xin-loading> -->
    </div>
  </div>
</template>

<script>
export default {
  components: {
  },
  data () {
    return {
      status: '',
      height: '50%'
    }
  },
  mounted () {
  },
  methods: {
    start () {
      this.$loading.start()
    },
    success () {
      this.$loading.success()
    },
    error () {
      this.$loading.error()
    }
  }
}
</script>
