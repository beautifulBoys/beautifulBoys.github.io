# 使用方法

# xin-ui

# 整体引用

import xinUI from 'xin-ui'
import 'xin-ui/styles.css'
Vue.use(xinUI)

# 单组件引用

import { Button, Checkbox } from 'xin-ui'
// 样式文件不提供单独打包。
import 'xin-ui/styles.css'
Vue.use(Button)
Vue.use(Checkbox)

# xin-skill

# 组件引用

import { Upload, Crop } from 'xin-ui/skill'
Vue.use(Upload)
Vue.use(Crop)
